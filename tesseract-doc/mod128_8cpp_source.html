00066 DIR128::DIR128(                 //from fcoord
00067                const FCOORD fc  //vector to quantize
00068               ) {
00069   int high, low, current;        //binary search
00070 
00071   low = 0;
00072   if (fc.y () == 0) {
00073     if (fc.x () >= 0)
00074       dir = 0;
00075     else
00076       dir = MODULUS / 2;
00077     return;
00078   }
00079   high = MODULUS;
00080   do {
00081     current = (high + low) / 2;
00082     if (dirtab[current] * fc >= 0)
00083       low = current;
00084     else
00085       high = current;
00086   }
00087   while (high - low > 1);
00088   dir = low;
00089 }
