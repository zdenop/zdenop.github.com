00590   // if replacements are made the length of best_choice might change.

00593     const UnicharAmbigsVector &table = replace ?
00594       getUnicharAmbigs().replace_ambigs() : getUnicharAmbigs().dang_ambigs();

00596       // Initialize ambig_blob_choices with lists containing a single
00597       // unichar id for the correspoding position in best_choice.
00598       // best_choice consisting from only the original letters will
00599       // have a rating of 0.0.

00601         BLOB_CHOICE_LIST *lst = new BLOB_CHOICE_LIST();
00602         BLOB_CHOICE_IT lst_it(lst);
00603         // TODO(rays/antonova) Should these BLOB_CHOICEs use real xheights
00604         // or are these fake ones good enough?

00606                                           0.0, 0.0, -1, -1, -1, 0, 1, false));
00607         ambig_blob_choices.push_back(lst);
00608       }
00609     }

00611     int wrong_ngram_index;

00615       if (i > 0) blob_index += best_choice->fragment_length(i-1);
00616       UNICHAR_ID curr_unichar_id = best_choice->unichar_id(i);
00617       if (stopper_debug_level > 2) {
00618         tprintf("Looking for %s ngrams starting with %s:\n",
00619                 replace ? "replaceable" : "ambiguous",
00620                 getUnicharset().debug_str(curr_unichar_id).string());
00621       }
00622       wrong_ngram_index = 0;
00623       wrong_ngram[wrong_ngram_index] = curr_unichar_id;
00624       if (curr_unichar_id == INVALID_UNICHAR_ID ||
00625           curr_unichar_id >= table.size() ||
00626           table[curr_unichar_id] == NULL) {
00627         continue;  // there is no ambig spec for this unichar id
00628       }

00630       for (spec_it.mark_cycle_pt(); !spec_it.cycled_list();) {
00631         const AmbigSpec *ambig_spec = spec_it.data();

00634                                                    ambig_spec->wrong_ngram);

00639           UnicharIdArrayUtils::print(ambig_spec->wrong_ngram, getUnicharset());
00640           tprintf("comparison result: %d\n", compare);
00641         }
00642         if (compare == 0) {
00643           // Record the place where we found an ambiguity.
00644           if (fixpt != NULL) {
00645             fixpt->push_back(DANGERR_INFO(
00646                 blob_index, blob_index+wrong_ngram_index, replace,
00647                 getUnicharset().get_isngram(ambig_spec->correct_ngram_id)));
00648             if (stopper_debug_level > 1) {
00649               tprintf("fixpt+=(%d %d %d %d)\n", blob_index,
00650                       blob_index+wrong_ngram_index, false,
00651                       getUnicharset().get_isngram(
00652                           ambig_spec->correct_ngram_id));
00653             }
00654           }
00655 
00656           if (replace) {
00657             if (stopper_debug_level > 2) {
00658               tprintf("replace ambiguity with: ");

00660                   ambig_spec->correct_fragments, getUnicharset());
00661             }

00663                          ambig_spec->correct_ngram_id,
00664                          best_choice, blob_choices, modified_blobs);
00665             modified_best_choice = true;
00666           } else if (i > 0 || ambig_spec->type != CASE_AMBIG) {
00667             // We found dang ambig - update ambig_blob_choices.
00668             if (stopper_debug_level > 2) {
00669               tprintf("found ambiguity: ");
00670               UnicharIdArrayUtils::print(
00671                   ambig_spec->correct_fragments, getUnicharset());
00672             }
00673             ambigs_found = true;
00674             for (int tmp_index = 0; tmp_index <= wrong_ngram_index;
00675                  ++tmp_index) {
00676               // Add a blob choice for the corresponding fragment of the
00677               // ambiguity. These fake blob choices are initialized with
00678               // negative ratings (which are not possible for real blob
00679               // choices), so that dawg_permute_and_select() considers any
00680               // word not consisting of only the original letters a better
00681               // choice and stops searching for alternatives once such a
00682               // choice is found.
00683               BLOB_CHOICE_IT bc_it(ambig_blob_choices[i+tmp_index]);
00684               bc_it.add_to_end(new BLOB_CHOICE(
00685                   ambig_spec->correct_fragments[tmp_index], -1.0, 0.0,
00686                   -1, -1, -1, 0, 1, false));
00687             }
00688           }
00689           spec_it.forward();
00690         } else if (compare == -1) {
00691           if (wrong_ngram_index+1 < ambig_spec->wrong_ngram_size &&
00692               ((next_index = wrong_ngram_index+1+i) < best_choice->length())) {
00693             // Add the next unichar id to wrong_ngram and keep looking for
00694             // more ambigs starting with curr_unichar_id in AMBIG_SPEC_LIST.
00695             wrong_ngram[++wrong_ngram_index] =
00696               best_choice->unichar_id(next_index);
00697           } else {
00698             break;  // no more matching ambigs in this AMBIG_SPEC_LIST
00699           }
00700         } else {
00701           spec_it.forward();
00702         }
00703       }  // end searching AmbigSpec_LIST
00704     }  // end searching best_choice
00705   }  // end searching replace and dangerous ambigs
00706 
00707   // If any ambiguities were found permute the constructed ambig_blob_choices
00708   // to see if an alternative dictionary word can be found.
00709   if (ambigs_found) {
00710     if (stopper_debug_level > 2) {
00711       tprintf("\nResulting ambig_blob_choices:\n");
00712       for (i = 0; i < ambig_blob_choices.length(); ++i) {
00713         print_ratings_list("", ambig_blob_choices.get(i), getUnicharset());
00714         tprintf("\n");
00715       }
00716     }
00717     WERD_CHOICE *alt_word = dawg_permute_and_select(ambig_blob_choices, 0.0);
00718     ambigs_found = (alt_word->rating() < 0.0);
00719     if (ambigs_found) {
00720       if (stopper_debug_level >= 1) {
00721         tprintf ("Stopper: Possible ambiguous word = %s\n",
00722                  alt_word->debug_string().string());
00723       }
00724       if (fixpt != NULL) {
00725         // Note: Currently character choices combined from fragments can only
00726         // be generated by NoDangrousAmbigs(). This code should be updated if
00727         // the capability to produce classifications combined from character
00728         // fragments is added to other functions.
00729         int orig_i = 0;
00730         for (i = 0; i < alt_word->length(); ++i) {
00731           if (alt_word->fragment_length(i) > 1) {
00732             fixpt->push_back(DANGERR_INFO(
00733                 orig_i, orig_i+alt_word->fragment_length(i)-1, true,
00734                 getUnicharset().get_isngram(alt_word->unichar_id(i))));
00735             if (stopper_debug_level > 1) {
00736               tprintf("fixpt->dangerous+=(%d %d %d %d)\n", orig_i,
00737                       (orig_i+alt_word->fragment_length(i)-1), true,
00738                       getUnicharset().get_isngram(alt_word->unichar_id(i)));
00739             }
00740           }
00741           orig_i += alt_word->fragment_length(i);
00742         }
00743       }
00744     }
00745     delete alt_word;
00746   }
00747   if (output_ambig_words_file_ != NULL) {
00748     fprintf(output_ambig_words_file_, "\n");
00749   }
00750 
00751   ambig_blob_choices.delete_data_pointers();
00752   return !ambigs_found;

00779 void Dict::ReplaceAmbig(int wrong_ngram_begin_index, int wrong_ngram_size,
00780                         UNICHAR_ID correct_ngram_id, WERD_CHOICE *werd_choice,
00781                         BLOB_CHOICE_LIST_VECTOR *blob_choices,
00782                         bool *modified_blobs) {
00783   int num_blobs_to_replace = 0;
00784   int begin_blob_index = 0;
00785   int i;
00786   for (i = 0; i < wrong_ngram_begin_index + wrong_ngram_size; ++i) {
00787     if (i >= wrong_ngram_begin_index) {
00788       num_blobs_to_replace +=  werd_choice->fragment_length(i);
00789     } else {
00790       begin_blob_index += werd_choice->fragment_length(i);
00791     }
00792   }
00793   BLOB_CHOICE_IT bit;
00794   int temp_blob_index = begin_blob_index;
00795   const char *temp_uch = NULL;
00796   const char *correct_ngram_str =
00797     getUnicharset().id_to_unichar(correct_ngram_id);
00798   for (int replaced_count = 0; replaced_count < wrong_ngram_size;
00799        ++replaced_count) {
00800     if (blob_choices != NULL) {
00801       UNICHAR_ID uch_id = werd_choice->unichar_id(wrong_ngram_begin_index);
00802       int fraglen = werd_choice->fragment_length(wrong_ngram_begin_index);
00803       if (fraglen > 1) temp_uch = getUnicharset().id_to_unichar(uch_id);
00804       for (i = 0; i < fraglen; ++i) {
00805         if (fraglen > 1) {
00806           STRING frag_str =
00807             CHAR_FRAGMENT::to_string(temp_uch, i, fraglen, false);
00808           getUnicharset().unichar_insert(frag_str.string());
00809           uch_id = getUnicharset().unichar_to_id(frag_str.string());
00810         }
00811         bit.set_to_list(blob_choices->get(temp_blob_index));
00812         STRING correct_frag_uch =
00813           CHAR_FRAGMENT::to_string(correct_ngram_str,
00814                                    temp_blob_index - begin_blob_index,
00815                                    num_blobs_to_replace, false);
00816         getUnicharset().unichar_insert(correct_frag_uch.string());
00817         UNICHAR_ID correct_frag_uch_id =
00818           getUnicharset().unichar_to_id(correct_frag_uch.string());
00819         // Find the WERD_CHOICE corresponding to the original unichar in
00820         // the list of blob choices, add the derived character fragment
00821         // before it with the same rating and certainty.
00822         for (bit.mark_cycle_pt(); !bit.cycled_list(); bit.forward()) {
00823           if (bit.data()->unichar_id() == correct_frag_uch_id) {
00824             break;  // the unichar we want to insert is already there
00825           }
00826           if (bit.data()->unichar_id() == uch_id) {
00827             bit.add_before_then_move(new BLOB_CHOICE(*(bit.data())));
00828             bit.data()->set_unichar_id(correct_frag_uch_id);
00829             if (modified_blobs != NULL) *modified_blobs = true;
00830             break;
00831           }
00832         }
00833         temp_blob_index++;
00834       }
00835     }
00836     // Remove current unichar from werd_choice. On the last iteration
00837     // set the correct replacement unichar instead of removing a unichar.
00838     if (replaced_count + 1 == wrong_ngram_size) {
00839       werd_choice->set_unichar_id(correct_ngram_id,
00840           num_blobs_to_replace, 0.0, 0.0, wrong_ngram_begin_index);
00841     } else {
00842       werd_choice->remove_unichar_id(wrong_ngram_begin_index);
00843     }
00844   }
00845   if (stopper_debug_level >= 1 && modified_blobs != NULL &&
00846       *modified_blobs && blob_choices != NULL) {
00847       werd_choice->print("ReplaceAmbig() ");
00848       tprintf("Modified blob_choices: ");
00849       for (int i = 0; i < blob_choices->size(); ++i) {
00850         print_ratings_list("\n", blob_choices->get(i), getUnicharset());
00851     }
00852   }

00860 int Dict::LengthOfShortestAlphaRun(const WERD_CHOICE &WordChoice) {
00861   int shortest = MAX_INT32;

00866     } else if (curr_len > 0) {

00869     }
00870   }

00872     shortest = curr_len;
00873   } else if (shortest == MAX_INT32) {
00874     shortest = 0;
00875   }
00876   return shortest;

00963 int Dict::UniformCertainties(const BLOB_CHOICE_LIST_VECTOR &Choices,
00964                              const WERD_CHOICE &BestChoice) {
00965   float Certainty;
00966   float WorstCertainty = MAX_FLOAT32;
00967   float CertaintyThreshold;
00968   FLOAT64 TotalCertainty;
00969   FLOAT64 TotalCertaintySquared;
00970   FLOAT64 Variance;
00971   FLOAT32 Mean, StdDev;
00972   int WordLength;
00973 
00974   WordLength = Choices.length();
00975   if (WordLength < 3)
00976     return true;
00977 
00978   TotalCertainty = TotalCertaintySquared = 0.0;
00979   BLOB_CHOICE_IT BlobChoiceIt;
00980   for (int i = 0; i < Choices.length(); ++i) {
00981     BlobChoiceIt.set_to_list(Choices.get(i));
00982     Certainty = BlobChoiceIt.data()->certainty();
00983     TotalCertainty += Certainty;
00984     TotalCertaintySquared += Certainty * Certainty;
00985     if (Certainty < WorstCertainty)
00986       WorstCertainty = Certainty;
00987   }
00988 
00989   // Subtract off worst certainty from statistics.
00990   WordLength--;
00991   TotalCertainty -= WorstCertainty;
00992   TotalCertaintySquared -= WorstCertainty * WorstCertainty;
00993 
00994   Mean = TotalCertainty / WordLength;
00995   Variance = ((WordLength * TotalCertaintySquared -
00996     TotalCertainty * TotalCertainty) /
00997     (WordLength * (WordLength - 1)));
00998   if (Variance < 0.0)
00999     Variance = 0.0;
01000   StdDev = sqrt (Variance);
01001 
01002   CertaintyThreshold = Mean - stopper_allowable_character_badness * StdDev;
01003   if (CertaintyThreshold > stopper_nondict_certainty_base)
01004     CertaintyThreshold = stopper_nondict_certainty_base;
01005 
01006   if (BestChoice.certainty() < CertaintyThreshold) {
01007     if (stopper_debug_level >= 1)
01008       cprintf("Stopper: Non-uniform certainty = %4.1f"
01009               " (m=%4.1f, s=%4.1f, t=%4.1f)\n",
01010               BestChoice.certainty(), Mean, StdDev, CertaintyThreshold);
01011     return false;
01012   } else {
01013     return true;
01014   }
01015 }
