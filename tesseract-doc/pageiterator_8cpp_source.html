00113 // Moves to the start of the next object at the given level in the
00114 // page hierarchy, and returns false if the end of the page was reached.
00115 // NOTE (CHANGED!) that ALL PageIteratorLevel level values will visit each
00116 // non-text block at least once.
00117 // Think of non text blocks as containing a single para, with at least one
00118 // line, with a single imaginary word, containing a single symbol.
00119 // The bounding boxes mark out any polygonal nature of the block, and
00120 // PTIsTextType(BLockType()) is false for non-text blocks.
00121 // Calls to Next with different levels may be freely intermixed.
00122 // This function iterates words in right-to-left scripts correctly, if
00123 // the appropriate language has been loaded into Tesseract.
00124 bool PageIterator::Next(PageIteratorLevel level) {
00125   if (it_->block() == NULL) return false;  // Already at the end!
00126   if (it_->word() == NULL)
00127     level = RIL_BLOCK;
00128 
00129   switch (level) {
00130     case RIL_BLOCK:
00131       it_->forward_block();
00132       break;
00133     case RIL_PARA:
00134       it_->forward_paragraph();
00135       break;
00136     case RIL_TEXTLINE:
00137       for (it_->forward_with_empties(); it_->row() == it_->prev_row();
00138            it_->forward_with_empties());
00139       break;
00140     case RIL_WORD:
00141       it_->forward_with_empties();
00142       break;
00143     case RIL_SYMBOL:
00144       if (cblob_it_ != NULL)
00145         cblob_it_->forward();
00146       ++blob_index_;
00147       if (blob_index_ >= word_length_)
00148         it_->forward_with_empties();
00149       else
00150         return true;
00151       break;
00152   }
00153   BeginWord(0);
00154   return it_->block() != NULL;
00155 }

00157 // Returns true if the iterator is at the start of an object at the given
00158 // level. Possible uses include determining if a call to Next(RIL_WORD)
00159 // moved to the start of a RIL_PARA.
00160 bool PageIterator::IsAtBeginningOf(PageIteratorLevel level) const {
00161   if (it_->block() == NULL) return false;  // Already at the end!
00162   if (it_->word() == NULL) return true;  // In an image block.
00163   switch (level) {
00164     case RIL_BLOCK:
00165       return blob_index_ == 0 && it_->block() != it_->prev_block();
00166     case RIL_PARA:
00167       return blob_index_ == 0 &&
00168           (it_->block() != it_->prev_block() ||
00169            it_->row()->row->para() != it_->prev_row()->row->para());
00170     case RIL_TEXTLINE:
00171       return blob_index_ == 0 && it_->row() != it_->prev_row();
00172     case RIL_WORD:
00173       return blob_index_ == 0;
00174     case RIL_SYMBOL:
00175       return true;
00176   }
00177   return false;
00178 }

00180 // Returns whether the iterator is positioned at the last element in a
00181 // given level. (e.g. the last word in a line, the last line in a block)
00182 bool PageIterator::IsAtFinalElement(PageIteratorLevel level,
00183                                     PageIteratorLevel element) const {
00184   if (Empty(element)) return true;  // Already at the end!
00185   // The result is true if we step forward by element and find we are
00186   // at the the end of the page or at beginning of *all* levels in:
00187   // [level, element).
00188   // When there is more than one level difference between element and level,
00189   // we could for instance move forward one symbol and still be at the first
00190   // word on a line, so we also have to be at the first symbol in a word.
00191   PageIterator next(*this);
00192   next.Next(element);
00193   if (next.Empty(element)) return true;  // Reached the end of the page.
00194   while (element > level) {
00195     element = static_cast<PageIteratorLevel>(element - 1);
00196     if (!next.IsAtBeginningOf(element))
00197       return false;
00198   }
00199   return true;
00200 }

00217 // ============= Accessing data ==============.
00218 // Coordinate system:
00219 // Integer coordinates are at the cracks between the pixels.
00220 // The top-left corner of the top-left pixel in the image is at (0,0).
00221 // The bottom-right corner of the bottom-right pixel in the image is at
00222 // (width, height).
00223 // Every bounding box goes from the top-left of the top-left contained
00224 // pixel to the bottom-right of the bottom-right contained pixel, so
00225 // the bounding box of the single top-left pixel in the image is:
00226 // (0,0)->(1,1).

00230 // Returns the bounding rectangle of the current object at the given level in
00231 // the coordinates of the working image that is pix_binary().
00232 // See comment on coordinate system above.
00233 // Returns false if there is no such object at the current position.
00234 bool PageIterator::BoundingBoxInternal(PageIteratorLevel level,
00235                                        int* left, int* top,
00236                                        int* right, int* bottom) const {

00238     return false;

00240   PARA *para = NULL;
00241   switch (level) {
00242     case RIL_BLOCK:
00243       box = it_->block()->block->bounding_box();
00244       break;
00245     case RIL_PARA:
00246       para = it_->row()->row->para();
00247       // explicit fall-through.
00248     case RIL_TEXTLINE:
00249       box = it_->row()->row->bounding_box();
00250       break;
00251     case RIL_WORD:
00252       box = it_->word()->word->bounding_box();
00253       break;
00254     case RIL_SYMBOL:
00255       if (cblob_it_ == NULL)
00256         box = it_->word()->box_word->BlobBox(blob_index_);
00257       else
00258         box = cblob_it_->data()->bounding_box();
00259   }
00260   if (level == RIL_PARA) {
00261     PageIterator other = *this;
00262     other.Begin();
00263     do {
00264       if (other.it_->block() &&
00265           other.it_->block()->block == it_->block()->block &&
00266           other.it_->row() && other.it_->row()->row &&
00267           other.it_->row()->row->para() == para) {
00268         box = box.bounding_union(other.it_->row()->row->bounding_box());
00269       }
00270     } while (other.Next(RIL_TEXTLINE));
00271   }
00272   if (level != RIL_SYMBOL || cblob_it_ != NULL)
00273     box.rotate(it_->block()->block->re_rotation());
00274   // Now we have a box in tesseract coordinates relative to the image rectangle,
00275   // we have to convert the coords to a top-down system.
00276   const int pix_height = pixGetHeight(tesseract_->pix_binary());
00277   const int pix_width = pixGetWidth(tesseract_->pix_binary());
00278   *left = ClipToRange(static_cast<int>(box.left()), 0, pix_width);
00279   *top = ClipToRange(pix_height - box.top(), 0, pix_height);
00280   *right = ClipToRange(static_cast<int>(box.right()), *left, pix_width);
00281   *bottom = ClipToRange(pix_height - box.bottom(), *top, pix_height);
00282   return true;
00283 }

00324 // Returns a binary image of the current object at the given level.
00325 // The position and size match the return from BoundingBoxInternal, and so this
00326 // could be upscaled with respect to the original input image.
00327 // Use pixDestroy to delete the image after use.
00328 // The following methods are used to generate the images:
00329 // RIL_BLOCK: mask the page image with the block polygon.
00330 // RIL_TEXTLINE: Clip the rectangle of the line box from the page image.
00331 // TODO(rays) fix this to generate and use a line polygon.
00332 // RIL_WORD: Clip the rectangle of the word box from the page image.
00333 // RIL_SYMBOL: Render the symbol outline to an image for cblobs (prior
00334 // to recognition) or the bounding box otherwise.
00335 // A reconstruction of the original image (using xor to check for double
00336 // representation) should be reasonably accurate,
00337 // apart from removed noise, at the block level. Below the block level, the
00338 // reconstruction will be missing images and line separators.
00339 // At the symbol level, kerned characters will be invade the bounding box
00340 // if rendered after recognition, making an xor reconstruction inaccurate, but
00341 // an or construction better. Before recognition, symbol-level reconstruction
00342 // should be good, even with xor, since the images come from the connected
00343 // components.
00344 Pix* PageIterator::GetBinaryImage(PageIteratorLevel level) const {
00345   int left, top, right, bottom;
00346   if (!BoundingBoxInternal(level, &left, &top, &right, &bottom))
00347     return NULL;
00348   Pix* pix = NULL;
00349   switch (level) {
00350     case RIL_BLOCK:
00351     case RIL_PARA:
00352       int bleft, btop, bright, bbottom;
00353       BoundingBoxInternal(RIL_BLOCK, &bleft, &btop, &bright, &bbottom);
00354       pix = it_->block()->block->render_mask();
00355       // AND the mask and the image.
00356       pixRasterop(pix, 0, 0, pixGetWidth(pix), pixGetHeight(pix),
00357                   PIX_SRC & PIX_DST, tesseract_->pix_binary(),
00358                   bleft, btop);
00359       if (level == RIL_PARA) {
00360         // RIL_PARA needs further attention:
00361         //   clip the paragraph from the block mask.
00362         Box* box = boxCreate(left - bleft, top - btop,
00363                              right - left, bottom - top);
00364         Pix* pix2 = pixClipRectangle(pix, box, NULL);
00365         boxDestroy(&box);
00366         pixDestroy(&pix);
00367         pix = pix2;
00368       }
00369       break;
00370     case RIL_TEXTLINE:
00371     case RIL_WORD:
00372     case RIL_SYMBOL:
00373       if (level == RIL_SYMBOL && cblob_it_ != NULL &&
00374           cblob_it_->data()->area() != 0)
00375         return cblob_it_->data()->render();
00376       // Just clip from the bounding box.
00377       Box* box = boxCreate(left, top, right - left, bottom - top);
00378       pix = pixClipRectangle(tesseract_->pix_binary(), box, NULL);
00379       boxDestroy(&box);
00380       break;
00381   }
00382   return pix;
00383 }

00512 // Sets up the internal data for iterating the blobs of a new word, then
00513 // moves the iterator to the given offset.

00515   WERD_RES* word_res = it_->word();

00518     word_length_ = 0;
00519     blob_index_ = 0;
00520     word_ = NULL;
00521     return;

00525     // is already baseline denormalized.

00537     if (cblob_it_ != NULL) {
00538       delete cblob_it_;

00540     }
00541   } else {
