00150   // Parse a comma-separated list of arguments into ArrayLists of the
00151   // possible types. Each type is stored in order, but the order
00152   // distinction between types is lost.
00153   // Note that the format is highly constrained to what the client used
00154   // to send to LUA:
00155   // Quoted string -> String.
00156   // true or false -> Boolean.
00157   // %f format number -> Float (no %e allowed)
00158   // Sequence of digits -> Integer
00159   // Nothing else allowed.
00160   private static void parseArguments(String argList,
00161                                      ArrayList<Integer> intList,
00162                                      ArrayList<Float> floatList,
00163                                      ArrayList<String> stringList,
00164                                      ArrayList<Boolean> boolList) {
00165     // str is only non-null if an argument starts with a single or double
00166     // quote. str is set back to null on completion of the string with a
00167     // matching quote. If the string contains a comma then str will stay
00168     // non-null across multiple argStr values until a matching closing quote.
00169     // Backslash escaped quotes do not count as terminating the string.
00170     String str = null;
00171     for (String argStr : argList.split(",")) {
00172       if (str != null) {
00173         // Last string was incomplete. Append argStr to it and restore comma.
00174         // Execute str += "," + argStr in Java.
00175         int length = str.length() + 1 + argStr.length();
00176         StringBuilder appended = new StringBuilder(length);
00177         appended.append(str);
00178         appended.append(",");
00179         appended.append(argStr);
00180         str =  appended.toString();
00181       } else if (argStr.length() == 0) {
00182         continue;
00183       } else {
00184         char quote = argStr.charAt(0);
00185         // If it begins with a quote then it is a string, but may not
00186         // end this time if it contained a comma.
00187         if (quote == '\'' || quote == '"') {
00188           str = argStr;
00189         }
00190       }
00191       if (str != null) {
00192         // It began with a quote. Check that it still does.
00193         assert str.charAt(0) == '\'' || str.charAt(0) == '"';
00194         int len = str.length();
00195         if (len > 1 && str.charAt(len - 1) == str.charAt(0)) {
00196           // We have an ending quote of the right type. Now check that
00197           // it is not escaped. Must have an even number of slashes before.
00198           int slash = len - 1;
00199           while (slash > 0 && str.charAt(slash - 1) == '\\')
00200             --slash;
00201           if ((len - 1 - slash) % 2 == 0) {
00202             // It is now complete. Chop off the quotes and save.
00203             // TODO(rays) remove the first backslash of each pair.
00204             stringList.add(str.substring(1, len - 1));
00205             str = null;
00206           }
00207         }
00208         // If str is not null here, then we have a string with a comma in it.
00209         // Append , and the next argument at the next iteration, but check
00210         // that str is null after the loop terminates in case it was an
00211         // unterminated string.
00212       } else if (floatPattern.matcher(argStr).matches()) {
00213         // It is a float.
00214         floatList.add(Float.parseFloat(argStr));
00215       } else if (argStr.equals("true")) {
00216         boolList.add(true);
00217       } else if (argStr.equals("false")) {
00218         boolList.add(false);
00219       } else if (intPattern.matcher(argStr).matches()) {
00220         // Only contains digits so must be an int.
00221         intList.add(Integer.parseInt(argStr));
00222       }
00223       // else ignore all incompatible arguments for forward compatibility.
00224     }
00225     // All strings must have been terminated.
00226     assert str == null;
00227   }
