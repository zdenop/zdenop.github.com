00047  * array_insert

00049  * Insert a data element into a particular spot in the array.  Move all
00050  * the elements in the array (past that spot) down one to make room for
00051  * the new element.
00052  **********************************************************************/
00053 ARRAY array_insert(ARRAY array, int index, void *value) { 
00054   int x;
00055 
00056   array = array_push (array, NULL);
00057   for (x = array_count (array) - 1; x > index; x--)
00058     array_value (array, x) = array_value (array, x - 1);
