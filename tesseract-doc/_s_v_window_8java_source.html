00531   private static String convertIntegerStringToUnicodeString(String input) {
00532     StringBuffer sb = new StringBuffer(input);
00533     Pattern numbers = Pattern.compile("0x[0-9a-fA-F]{4}");
00534     Matcher matcher = numbers.matcher(sb);
00535 
00536     while (matcher.find()) {
00537       // Find the next match which resembles a hexadecimal value and convert it
00538       // to
00539       // its char value
00540       char a = (char) (Integer.decode(matcher.group()).intValue());
00541 
00542       // Replace the original with the new character
00543       sb.replace(matcher.start(), matcher.end(), String.valueOf(a));
00544 
00545       // Start again, since our positions have switched
00546       matcher.reset();
00547     }
00548     return sb.toString();
00549   }
